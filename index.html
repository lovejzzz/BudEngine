üß™ Testing API Example

// Get game state
const state = engine.test.getState();
console.log(state.player.health);

// Inject input
engine.test.input('w', true);  // hold W
engine.test.click(100, 200);   // click at world position

// Time control
engine.test.step(60);          // advance 60 frames (1 second)
engine.test.fastForward(5);    // skip 5 seconds instantly

// Run auto-playtest
const results = engine.test.autoplay({
    strategy: 'survive',  // or 'aggressive', 'explore', 'random'
    duration: 60,         // seconds
    runs: 50,             // number of playtests
    report: true
});

console.log('Avg survival:', results.avgSurvival);
console.log('Deaths:', results.deaths);
console.log('Balance notes:', results.balanceNotes);

üöÄ Quick Start

<!DOCTYPE html>
<html>
<head>
    <title>My Game</title>
</head>
<body>
    <script src="bud.js"></script>
    <script>
        const engine = new BudEngine({ width: 1280, height: 720 });

        engine.scene('game', {
            enter() {
                // Spawn player
                const player = engine.spawn('player', {
                    x: 640, y: 360,
                    sprite: engine.art.character({ 
                        body: 'circle', 
                        color: '#00ffcc', 
                        size: 20 
                    }),
                    collider: { type: 'circle', radius: 16 },
                    health: 100,
                    speed: 200
                });

                // Camera follows player
                engine.cameraFollow(player);
            },

            update(dt) {
                const player = engine.findOne('player');
                
                // Movement
                if (engine.input.key('w')) player.y -= player.speed * dt;
                if (engine.input.key('s')) player.y += player.speed * dt;
                if (engine.input.key('a')) player.x -= player.speed * dt;
                if (engine.input.key('d')) player.x += player.speed * dt;
            }
        });

        engine.goTo('game');
        engine.start();
    </script>
</body>
</html>
